  • UK_AnimData: к°Ғ л¬ҙкё°лі„ лӢӨлҘё мҪӨліҙ кіөкІ©мқ„ м •мқҳн•ҳм—¬ кҙҖлҰ¬

// PoseSearch мұ„л„җ к°ҖмӨ‘м№ҳ м„Өм • мҳҲмӢң
void UMyAnimInstance::UpdateMotionMatchingWeights()
{
    bool bInCombat = HasTag(Tag_State_Combat);

    // м „нҲ¬ мӢң мғҒмІҙ лҸҷмһ‘м—җ к°ҖмӨ‘м№ҳ мҰқк°Җ
    FPoseSearchChannelSettings Settings;
    Settings.PoseWeight    = bInCombat ? 0.4f : 0.6f;
    Settings.TrajectoryWeight = bInCombat ? 0.6f : 0.4f;

    SetMotionMatchingChannelWeights(Settings);
}
// Trajectory лӘ©н‘ң мғқм„ұ
FPoseSearchQueryTrajectorySample
UMyAnimInstance::BuildTrajectorySample(
    float DeltaTime) const
{
    FVector Velocity   = GetOwnerVelocity();
    FVector InputDir   = GetInputVector();

    // 0.5мҙҲ л’Ө мҳҲмёЎ мң„м№ҳ
    FVector FuturePos  = Velocity * 0.5f
                       + InputDir * 200.f;

    return { FuturePos, Velocity };
}

Ability System м„Өкі„

GASмқҳ AbilitySystemComponentлҘј PlayerStateм—җ л°°м№ҳн•ҳкі , лҰ¬мҠӨнҸ° мӢңм—җлҸ„ м–ҙл№ҢлҰ¬нӢ°В·м–ҙнҠёлҰ¬л·°нҠёк°Җ мң м§ҖлҗҳлҸ„лЎқ м„Өкі„н–ҲмҠөлӢҲлӢӨ.

// SetByCallerлҘј нҶөн•ң лҚ°лҜём§Җ м „лӢ¬
void UGA_MeleeAttack::ApplyDamage(
    AActor* Target)
{
    FGameplayEffectSpecHandle SpecHandle =
        MakeOutgoingGameplayEffectSpec(
            DamageEffectClass, GetAbilityLevel());

    // лҹ°нғҖмһ„м—җ лҚ°лҜём§Җ к°’ мЈјмһ…
    SpecHandle.Data->SetSetByCallerMagnitude(
        TAG_Data_Damage,
        BaseDamage * ComboMultiplier);

    ApplyGameplayEffectSpecToTarget(
        SpecHandle,
        UAbilitySystemGlobals::
        GetAbilitySystemComponentFromActor(Target));
}
// Tag кё°л°ҳ м–ҙл№ҢлҰ¬нӢ° мӢӨн–ү мЎ°кұҙ
bool UGA_MeleeAttack::CanActivateAbility(
    ...) const
{
    // Stun/Dead мғҒнғңл©ҙ мӮ¬мҡ© л¶Ҳк°Җ
    if (HasMatchingGameplayTag(
            TAG_State_Stunned) ||
        HasMatchingGameplayTag(
            TAG_State_Dead))
    {
        return false;
    }
    return Super::CanActivateAbility(...);
}

// мғҒнғң кё°л°ҳ мқҙлҸҷ мҶҚлҸ„ м„Өм •
void UMyCharMovement::UpdateMovementParams()
{
    const bool bSprinting =
        HasTag(TAG_State_Sprinting);
    const bool bInCombat  =
        HasTag(TAG_State_Combat);

    if (bSprinting)
        MaxWalkSpeed = SprintSpeed;     // 600
    else if (bInCombat)
        MaxWalkSpeed = CombatWalkSpeed; // 300
    else
        MaxWalkSpeed = DefaultSpeed;    // 450

    // м „нҲ¬ мӨ‘ нҡҢм „ мҰүмӢң л°ҳмқ‘
    bOrientRotationToMovement = !bInCombat;
    bUseControllerDesiredRotation = bInCombat;
}
